function plakaEkle(plaka){ plaka=plaka.toUpperCase().trim(); const ss=SpreadsheetApp.getActiveSpreadsheet(); let sh=ss.getSheetByName("Filo_Araclar"); if(!sh){ sh=ss.insertSheet("Filo_Araclar"); sh.appendRow(["ID","PLAKA"]); } const data=sh.getDataRange().getValues(); for(let i=1;i=1;i--){ if(data[i][1].toString().toUpperCase()==plaka){ sh.deleteRow(i+1); return "SİLİNDİ"; } } return "BULUNAMADI"; }
